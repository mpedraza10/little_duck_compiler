program BasicProgram;

var i, j: int;

main
{
    i = 5;
    j = 10;

    do {
        i = i + 1;

        if (i < j) {
            print("La i sigue siendo menor: i = ", i);
        } else {
            print("La i es ahora igual que j");
            print("i = ", i);
            print("j = ", j);
        };
    } while ( i < j );
}
end
